query<-"SELECT yearID,HR
FROM Batting
WHERE playerID='ruthba01'"

result<-sqldf(query)

ggplot()+
  geom_line(data=result,aes(x=yearID,y=HR),color="green")+
  ggtitle("Ruth's Homerun Totals Throughout the Year")+
  xlab("Year")+
  ylab("Homeruns")
